Компоновки:
размещение виджетов в окне

📐 Темы: QVBoxLayout, QHBoxLayout, QGridLayout, QStackedLayout, QTabWidget ⏱ ~40 мин

Зачем нужны компоновки?

До сих пор мы успешно создавали окна и добавляли в них один виджет. Однако в реальных приложениях требуется размещать сразу несколько элементов управления: кнопки, поля ввода, метки и т.д. Чтобы управлять их расположением и автоматически адаптировать интерфейс при изменении размеров окна, в Qt используются компоновки (layouts).

В Qt доступно четыре базовые компоновки:

КомпоновкаПоведение
QHBoxLayoutЛинейная горизонтальная компоновка (слева направо)
QVBoxLayoutЛинейная вертикальная компоновка (сверху вниз)
QGridLayoutСетка с индексируемыми строками и столбцами
QStackedLayoutСтопкой (по оси Z) — одновременно виден только один виджет

Компоновки можно вкладывать друг в друга, создавая сложные иерархические структуры интерфейса.

Вспомогательный виджет Color

Для наглядности примеров создадим простой пользовательский виджет, залитый сплошным цветом. Сохраните следующий код в файл layout_colorwidget.py:

layout_colorwidget.py 
from PySide6.QtGui import QColor, QPalette
from PySide6.QtWidgets import QWidget

class Color(QWidget):
    def __init__(self, color):
        super().__init__()
        self.setAutoFillBackground(True)
        palette = self.palette()
        palette.setColor(QPalette.Window, QColor(color))
        self.setPalette(palette)

Благодаря этому виджету мы сможем легко различать элементы компоновок: Color("red") создаст красный прямоугольник, Color("blue") — синий и т.д.

QVBoxLayout — вертикальное расположение

QVBoxLayout располагает виджеты один над другим. Каждый новый виджет добавляется в нижнюю часть столбца.

vertical_layout.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QVBoxLayout, QWidget
from layout_colorwidget import Color

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Вертикальная компоновка")
        
        layout = QVBoxLayout()
        layout.addWidget(Color("red"))
        layout.addWidget(Color("green"))
        layout.addWidget(Color("blue"))
        
        widget = QWidget()
        widget.setLayout(layout)
        self.setCentralWidget(widget)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
🎨 Результат: три цветных прямоугольника (красный, зелёный, синий) располагаются друг под другом, растягиваясь по ширине окна.

QHBoxLayout — горизонтальное расположение

QHBoxLayout размещает виджеты горизонтально, слева направо.

🎨 Результат: три цветных прямоугольника располагаются в ряд: красный, зелёный, синий.

Вложенные компоновки (Nesting Layouts)

Чтобы создавать сложные интерфейсы, компоновки можно вкладывать друг в друга с помощью метода .addLayout(). Следующий пример комбинирует вертикальные и горизонтальные компоновки.

nested_layout.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QHBoxLayout, QVBoxLayout, QWidget
from layout_colorwidget import Color

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Вложенные компоновки")
        
        main_layout = QHBoxLayout()
        
        left_column = QVBoxLayout()
        left_column.addWidget(Color("red"))
        left_column.addWidget(Color("yellow"))
        
        right_column = QVBoxLayout()
        right_column.addWidget(Color("purple"))
        right_column.addWidget(Color("blue"))
        
        main_layout.addLayout(left_column)
        main_layout.addWidget(Color("green"))
        main_layout.addLayout(right_column)
        
        widget = QWidget()
        widget.setLayout(main_layout)
        self.setCentralWidget(widget)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Настройка отступов и промежутков: layout.setContentsMargins(0,0,0,0) убирает внешние отступы, layout.setSpacing(20) задаёт расстояние между виджетами в пикселях.

QGridLayout — расположение в сетке

QGridLayout позволяет размещать виджеты в ячейках сетки, задавая номер строки и столбца. Это идеально подходит для создания форм и диалоговых окон.

grid_layout.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QGridLayout, QWidget
from layout_colorwidget import Color

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Сетка (Grid)")
        
        layout = QGridLayout()
        layout.addWidget(Color("red"), 0, 0)    # строка 0, столбец 0
        layout.addWidget(Color("green"), 1, 0)  # строка 1, столбец 0
        layout.addWidget(Color("blue"), 1, 1)   # строка 1, столбец 1
        layout.addWidget(Color("purple"), 2, 1) # строка 2, столбец 1
        
        widget = QWidget()
        widget.setLayout(layout)
        self.setCentralWidget(widget)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
🎨 Результат: красный прямоугольник в левом верхнем углу (0,0). Зелёный под ним (1,0). Синий справа от зелёного (1,1). Фиолетовый под синим (2,1). Пустая ячейка (0,1) остаётся незаполненной.

QStackedLayout и QStackedWidget — переключение виджетов

QStackedLayout (и его готовый аналог QStackedWidget) размещает виджеты друг над другом (по оси Z), показывая только один за раз. Это основа для интерфейсов с вкладками, мастеров установки и пошаговых диалогов.

stacked_with_buttons.py 
import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QHBoxLayout, QPushButton,
    QStackedLayout, QVBoxLayout, QWidget
)
from layout_colorwidget import Color

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("QStackedLayout")
        
        page_layout = QVBoxLayout()
        btn_layout = QHBoxLayout()
        self.stack = QStackedLayout()
        
        # Кнопки для переключения
        btn_red = QPushButton("Красный")
        btn_red.clicked.connect(lambda: self.stack.setCurrentIndex(0))
        btn_green = QPushButton("Зелёный")
        btn_green.clicked.connect(lambda: self.stack.setCurrentIndex(1))
        btn_blue = QPushButton("Синий")
        btn_blue.clicked.connect(lambda: self.stack.setCurrentIndex(2))
        
        btn_layout.addWidget(btn_red)
        btn_layout.addWidget(btn_green)
        btn_layout.addWidget(btn_blue)
        
        self.stack.addWidget(Color("red"))
        self.stack.addWidget(Color("green"))
        self.stack.addWidget(Color("blue"))
        
        page_layout.addLayout(btn_layout)
        page_layout.addLayout(self.stack)
        
        widget = QWidget()
        widget.setLayout(page_layout)
        self.setCentralWidget(widget)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

QTabWidget — готовые вкладки

Qt предоставляет удобный виджет QTabWidget, который реализует вкладки «из коробки»:

tabwidget_demo.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QTabWidget
from layout_colorwidget import Color

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Вкладки (QTabWidget)")
        
        tabs = QTabWidget()
        tabs.setTabPosition(QTabWidget.West)   # Вкладки слева
        tabs.setMovable(True)                  # Вкладки можно перетаскивать
        
        tabs.addTab(Color("red"), "Красный")
        tabs.addTab(Color("green"), "Зелёный")
        tabs.addTab(Color("blue"), "Синий")
        tabs.addTab(Color("yellow"), "Жёлтый")
        
        self.setCentralWidget(tabs)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
🎨 Результат: окно с четырьмя вкладками. Положение вкладок можно настроить с помощью setTabPosition() (North, South, West, East), а перетаскивание вкладок — методом setMovable(True).
Итог лекции: Компоновки — это фундамент построения интерфейсов. QVBoxLayout и QHBoxLayout создают линейные структуры, QGridLayout — таблицы (формы), а QStackedLayout/QTabWidget — интерфейсы с переключением страниц. Вложенность компоновок позволяет реализовать интерфейс любой сложности без жёсткой привязки к координатам.
