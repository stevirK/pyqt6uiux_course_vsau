Диалоговые окна:
QDialog и взаимодействие с пользователем

📋 Темы: QDialog, модальные окна, QDialogButtonBox, accept/reject ⏱ ~30 мин

Что такое диалоговые окна?

Диалоговые окна — это небольшие окна, которые используются для взаимодействия с пользователем: запрос подтверждения, ввод данных, выбор файла, настройки приложения. В Qt диалоговые окна реализуются классом QDialog.

Ключевая особенность большинства диалогов — модальность. Модальное диалоговое окно блокирует взаимодействие с родительским окном, пока не будет закрыто. Это заставляет пользователя ответить на вопрос диалога, прежде чем продолжить работу.

Важно: В Qt есть два основных типа модальности:
  • Модальное относительно приложения — блокирует все окна приложения
  • Модальное относительно окна — блокирует только родительское окно (наиболее распространено)

Простое диалоговое окно

Начнём с базового приложения, которое открывает пустой диалог при нажатии кнопки.

simple_dialog.py 
import sys
from PySide6.QtWidgets import QApplication, QDialog, QMainWindow, QPushButton

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Моё приложение")
        
        button = QPushButton("Открыть диалог")
        button.clicked.connect(self.button_clicked)
        self.setCentralWidget(button)
    
    def button_clicked(self):
        dlg = QDialog(self)           # Передаём self как родителя
        dlg.setWindowTitle("Диалог")
        dlg.exec()                    # Запускаем модальный диалог

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
🎨 Результат: при нажатии на кнопку появляется пустое диалоговое окно поверх главного окна. Пока диалог открыт, главное окно недоступно.

Создание собственного диалога (подкласс QDialog)

Для создания полноценных диалогов лучше наследоваться от QDialog и настраивать его в методе __init__. Добавим текстовую метку и кнопки OK/Cancel.

custom_dialog.py 
import sys
from PySide6.QtWidgets (
    QApplication, QDialog, QMainWindow, QPushButton,
    QVBoxLayout, QLabel, QDialogButtonBox
)

class CustomDialog(QDialog):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("Подтверждение")
        
        # Создаём содержимое диалога
        message = QLabel("Вы уверены, что хотите продолжить?")
        
        # Создаём кнопки OK и Cancel
        buttons = QDialogButtonBox.Ok | QDialogButtonBox.Cancel
        self.button_box = QDialogButtonBox(buttons)
        
        # Подключаем сигналы кнопок к методам accept/reject
        self.button_box.accepted.connect(self.accept)
        self.button_box.rejected.connect(self.reject)
        
        # Компоновка
        layout = QVBoxLayout()
        layout.addWidget(message)
        layout.addWidget(self.button_box)
        self.setLayout(layout)

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Моё приложение")
        
        button = QPushButton("Открыть диалог")
        button.clicked.connect(self.button_clicked)
        self.setCentralWidget(button)
    
    def button_clicked(self):
        dlg = CustomDialog(self)
        if dlg.exec():          # exec() возвращает QDialog.Accepted или QDialog.Rejected
            print("Пользователь нажал OK")
        else:
            print("Пользователь нажал Cancel или закрыл окно")

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Как работает QDialogButtonBox

QDialogButtonBox — это специальный виджет, который автоматически размещает стандартные кнопки диалога в правильном порядке для каждой платформы.

КнопкаОписание

QDialogButtonBox.CancelКнопка "Отмена"
QDialogButtonBox.YesКнопка "Да"

buttons = QDialogButtonBox.Yes | QDialogButtonBox.No | QDialogButtonBox.Cancel

Встроенные диалоги Qt

Qt предоставляет готовые диалоги для самых частых задач. Это избавляет вас от необходимости создавать их с нуля и обеспечивает нативный внешний вид.

QMessageBox — диалог с сообщением

messagebox_demo.py 
from PySide6.QtWidgets import QMessageBox

# Внутри метода MainWindow:
def show_message(self):
    reply = QMessageBox.question(
        self,
        "Подтверждение",
        "Вы действительно хотите выйти?",
        QMessageBox.Yes | QMessageBox.No,
        QMessageBox.No  # Кнопка по умолчанию
    )
    
    if reply == QMessageBox.Yes:
        print("Пользователь подтвердил")
        self.close()
    else:
        print("Отменено")

Типы сообщений в QMessageBox:

  • QMessageBox.information() — информационное сообщение (синий значок "i")
  • QMessageBox.warning() — предупреждение (жёлтый треугольник)
  • QMessageBox.critical() — критическая ошибка (красный крест)
  • QMessageBox.question() — вопрос с вариантами ответа
  • QMessageBox.about() — диалог "О программе"

QFileDialog — выбор файлов и папок

filedialog_demo.py 
from PySide6.QtWidgets import QFileDialog

# Открытие файла
def open_file(self):
    file_path, _ = QFileDialog.getOpenFileName(
        self,
        "Открыть файл",
        "",  # Начальная директория
        "Текстовые файлы (*.txt);;Все файлы (*.*)"
    )
    if file_path:
        print(f"Выбран файл: {file_path}")

# Сохранение файла
def save_file(self):
    file_path, _ = QFileDialog.getSaveFileName(
        self,
        "Сохранить файл",
        "безымянный.txt",
        "Текстовые файлы (*.txt)"
    )
    if file_path:
        print(f"Сохранение в: {file_path}")

# Выбор папки
def open_folder(self):
    folder_path = QFileDialog.getExistingDirectory(
        self,
        "Выбрать папку",
        ""
    )
    if folder_path:
        print(f"Выбрана папка: {folder_path}")

# Выбор нескольких файлов
def open_multiple_files(self):
    files, _ = QFileDialog.getOpenFileNames(
        self,
        "Выбрать файлы",
        "",
        "Изображения (*.png *.jpg *.bmp)"
    )
    print(f"Выбрано файлов: {len(files)}")

QInputDialog — простой ввод данных

inputdialog_demo.py 
from PySide6.QtWidgets import QInputDialog

# Ввод текста
def get_text(self):
    text, ok = QInputDialog.getText(
        self,
        "Ввод текста",
        "Введите ваше имя:"
    )
    if ok and text:
        print(f"Привет, {text}!")

# Ввод числа
def get_number(self):
    value, ok = QInputDialog.getInt(
        self,
        "Выбор числа",
        "Введите возраст:",
        min=0, max=120, step=1
    )
    if ok:
        print(f"Возраст: {value}")

# Выбор из списка
def choose_item(self):
    items = ["Красный", "Зелёный", "Синий"]
    item, ok = QInputDialog.getItem(
        self,
        "Выбор цвета",
        "Выберите цвет:",
        items,
        current=0,
        editable=False
    )
    if ok and item:
        print(f"Выбран цвет: {item}")

Полный пример: настройки приложения

settings_dialog.py 
import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QPushButton, QLabel, QVBoxLayout,
    QDialog, QDialogButtonBox, QLineEdit, QSpinBox, QCheckBox, QWidget
)

class SettingsDialog(QDialog):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("Настройки")
        
        # Поля ввода
        self.name_edit = QLineEdit()
        self.age_spin = QSpinBox()
        self.age_spin.setRange(0, 150)
        self.notify_check = QCheckBox("Уведомления")
        
        # Кнопки
        button_box = QDialogButtonBox(QDialogButtonBox.Ok | QDialogButtonBox.Cancel)
        button_box.accepted.connect(self.accept)
        button_box.rejected.connect(self.reject)
        
        # Компоновка
        layout = QVBoxLayout()
        layout.addWidget(QLabel("Имя:"))
        layout.addWidget(self.name_edit)
        layout.addWidget(QLabel("Возраст:"))
        layout.addWidget(self.age_spin)
        layout.addWidget(self.notify_check)
        layout.addWidget(button_box)
        self.setLayout(layout)
    
    def get_data(self):
        return {
            "name": self.name_edit.text(),
            "age": self.age_spin.value(),
            "notify": self.notify_check.isChecked()
        }

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Приложение с настройками")
        
        self.label = QLabel("Настройки не загружены")
        self.label.setAlignment(Qt.AlignCenter)
        
        button = QPushButton("Настройки")
        button.clicked.connect(self.open_settings)
        
        container = QWidget()
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        layout.addWidget(button)
        container.setLayout(layout)
        self.setCentralWidget(container)
    
    def open_settings(self):
        dlg = SettingsDialog(self)
        if dlg.exec():
            data = dlg.get_data()
            self.label.setText(
                f"Имя: {data['name']}, Возраст: {data['age']}, "
                f"Уведомления: {'Вкл' if data['notify'] else 'Выкл'}"
            )

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Итог лекции: Диалоговые окна — неотъемлемая часть любого серьёзного GUI-приложения. QDialog предоставляет гибкую основу для создания собственных диалогов, а встроенные классы (QMessageBox, QFileDialog, QColorDialog и др.) покрывают 90% повседневных задач. Передавайте parent в диалог, чтобы он автоматически центрировался относительно родительского окна.
