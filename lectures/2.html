Сигналы и слоты — сердце Qt

В прошлой лекции мы создали окно и добавили кнопку, но кнопка ничего не делала. Чтобы приложения оживали, нам нужен способ связать действия пользователя (нажатие, выбор, ввод) с реальным поведением программы. В Qt эта задача решается с помощью механизма сигналов и слотов.

Сигналы — это уведомления, которые виджеты отправляют при наступлении какого-либо события. Этим событием может быть что угодно: от нажатия кнопки до изменения текста в поле ввода. Многие сигналы инициируются действиями пользователя, но это не является правилом. Сигналы также могут передавать данные, предоставляя дополнительную информацию о произошедшем.

Слоты — это получатели сигналов. В Python любая функция или метод может использоваться в качестве слота — достаточно просто подключить к ней сигнал. Если сигнал отправляет данные, принимающая функция получит эти данные. Многие виджеты Qt имеют собственные встроенные слоты, что позволяет связывать виджеты напрямую друг с другом.

QPushButton — первые сигналы

Подключим сигнал clicked от кнопки к пользовательскому методу. Кнопку можно сделать переключаемой — это позволит отслеживать её состояние.

signals_slots.py 
from PySide6.QtWidgets import QApplication, QMainWindow, QPushButton
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Моё приложение")
        
        button = QPushButton("Нажми меня!")
        button.setCheckable(True)          # Переключаемый режим
        button.clicked.connect(self.the_button_was_clicked)
        button.clicked.connect(self.the_button_was_toggled)
        
        self.setCentralWidget(button)
    
    def the_button_was_clicked(self):
        print("Нажато!")
    
    def the_button_was_toggled(self, checked):
        print("Нажата?", checked)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Вы можете подключить сколько угодно слотов к одному сигналу. Сигнал .clicked передаёт состояние нажатия (для обычных кнопок — False, но для переключаемых — True/False).

Изменение интерфейса из слота

В слоте можно модифицировать не только саму кнопку, но и другие элементы интерфейса. Например, изменим текст кнопки после нажатия, отключим её и обновим заголовок окна.

modify_ui.py 
from PySide6.QtWidgets import QApplication, QMainWindow, QPushButton
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Моё приложение")
        
        self.button = QPushButton("Нажми меня!")
        self.button.clicked.connect(self.the_button_was_clicked)
        self.setCentralWidget(self.button)
    
    def the_button_was_clicked(self):
        self.button.setText("Вы уже нажали на меня.")
        self.button.setEnabled(False)
        self.setWindowTitle("Моё приложение с одним нажатием")

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Важно: Сигналы не ограничены только тем виджетом, который их инициировал. В слоте можно делать всё что угодно: менять другие виджеты, обновлять данные, вызывать функции.

Сигнал изменения заголовка окна

QMainWindow тоже имеет сигналы. Например, windowTitleChanged срабатывает при изменении заголовка окна (и только тогда, когда новый заголовок отличается от предыдущего). Это позволяет строить цепочки реакций.

window_title_demo.py 
from PySide6.QtWidgets import QApplication, QMainWindow, QPushButton
import sys
from random import choice

window_titles = [
    "My App", "Still My App", "What on earth",
    "This is surprising", "Something went wrong"
]

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        self.button = QPushButton("Press Me!")
        self.button.clicked.connect(self.the_button_was_clicked)
        self.windowTitleChanged.connect(self.the_window_title_changed)
        
        self.setCentralWidget(self.button)
    
    def the_button_was_clicked(self):
        new_title = choice(window_titles)
        self.setWindowTitle(new_title)
    
    def the_window_title_changed(self, window_title):
        if window_title == "Something went wrong":
            self.button.setDisabled(True)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Одно событие (нажатие кнопки) запускает цепочку изменений: меняется заголовок, а сигнал от заголовка управляет доступностью кнопки. Такое разделение ответственности — ключевой принцип GUI-приложений.

Обзор основных виджетов

Qt поставляется с большим набором готовых виджетов. Мы рассмотрим три важнейших: QLabel, QCheckBox и QComboBox.

QLabel — текстовая метка

QLabel — простейший виджет для отображения текста или изображения. Можно настраивать шрифт, выравнивание и содержимое.

label_demo.py 
from PySide6.QtWidgets import QApplication, QLabel, QMainWindow
from PySide6.QtCore import Qt
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        label = QLabel("Привет, мир!")
        font = label.font()
        font.setPointSize(30)
        label.setFont(font)
        label.setAlignment(Qt.AlignHCenter | Qt.AlignVCenter)
        
        self.setCentralWidget(label)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
ФлагПоведение (горизонтальное)
Qt.AlignLeftВыравнивание по левому краю
Qt.AlignRightВыравнивание по правому краю
Qt.AlignHCenterЦентрирование по горизонтали
Qt.AlignJustifyВыравнивание по ширине

Вертикальные флаги: AlignTop, AlignBottom, AlignVCenter. Флаги комбинируются через |.

QCheckBox — флажок

QCheckBox предоставляет переключаемый флажок. Может быть двухсостоянием (вкл/выкл) или трёхсостоянием (частично выбран). Сигнал stateChanged передаёт числовое состояние.

checkbox_demo.py 
from PySide6.QtWidgets import QApplication, QCheckBox, QMainWindow
from PySide6.QtCore import Qt
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        checkbox = QCheckBox("Это флажок")
        checkbox.setCheckState(Qt.Checked)   # или Qt.Unchecked / Qt.PartiallyChecked
        # checkbox.setTristate(True)         # включить трёхсостояние
        
        checkbox.stateChanged.connect(self.show_state)
        self.setCentralWidget(checkbox)
    
    def show_state(self, s):
        print(s == Qt.Checked)   # True/False
        print(s)                 # 2, 0 или 1

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Состояния QCheckBox:

  • Qt.Checked (2) — включён
  • Qt.Unchecked (0) — выключен
  • Qt.PartiallyChecked (1) — частично (трёхсостояние)

QComboBox — раскрывающийся список

QComboBox позволяет выбирать один элемент из выпадающего списка. Его можно сделать редактируемым (пользователь вводит своё значение) и ограничивать количество элементов.

combobox_demo.py 
from PySide6.QtWidgets import QApplication, QComboBox, QMainWindow
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        combo = QComboBox()
        combo.addItems(["Один", "Два", "Три"])
        combo.currentIndexChanged.connect(self.index_changed)
        combo.currentTextChanged.connect(self.text_changed)
        
        # Дополнительные настройки:
        # combo.setEditable(True)                    # возможность ввода своего варианта
        # combo.setInsertPolicy(QComboBox.InsertAlphabetically)
        # combo.setMaxCount(10)
        
        self.setCentralWidget(combo)
    
    def index_changed(self, i):
        print(f"Индекс: {i}")
    
    def text_changed(self, s):
        print(f"Текст: {s}")

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Политика вставки (InsertPolicy)Описание
QComboBox.NoInsertНе вставлять пользовательский текст
QComboBox.InsertAtTopВставить в начало списка
QComboBox.InsertAtBottomВставить в конец
QComboBox.InsertAlphabeticallyВставить в алфавитном порядке
