Панели инструментов (Toolbars)

Панели инструментов — это полосы с иконками и/или текстом, используемые для быстрого доступа к типовым задачам приложения. В Qt панели инструментов реализованы классом QToolBar. Добавить панель в QMainWindow очень просто:

simple_toolbar.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QLabel, QToolBar
from PySide6.QtCore import Qt

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        label = QLabel("Привет!")
        label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(label)
        
        toolbar = QToolBar("Моя главная панель")
        self.addToolBar(toolbar)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Важно: Панели инструментов в QMainWindow можно перетаскивать, закреплять в разных местах и отключать. Это стандартное поведение Qt.

QAction — абстрактное действие

QAction — один из самых полезных классов в Qt. Он позволяет описать действие один раз, а затем использовать его в разных местах интерфейса: на панели инструментов, в меню, с горячей клавишей. Одно действие — много представлений.

Основные возможности QAction:

  • Текст (setText()) и иконка (setIcon())
  • Подсказка статуса (setStatusTip())
  • Горячая клавиша (setShortcut())
  • Переключаемый режим (setCheckable())
  • Сигнал .triggered — возникает при активации действия
action_demo.py 
import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QLabel, QToolBar, QStatusBar
)
from PySide6.QtCore import Qt
from PySide6.QtGui import QAction

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        label = QLabel("Привет!")
        label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(label)
        
        # Панель инструментов
        toolbar = QToolBar("Главная панель")
        self.addToolBar(toolbar)
        
        # Создаём действие
        button_action = QAction("Моя кнопка", self)
        button_action.setStatusTip("Это подсказка статуса")
        button_action.triggered.connect(self.on_button_click)
        button_action.setCheckable(True)   # переключаемый режим
        
        # Добавляем действие на панель инструментов
        toolbar.addAction(button_action)
        
        # Строка состояния
        self.setStatusBar(QStatusBar(self))
    
    def on_button_click(self, checked):
        print("Нажато! Состояние переключателя:", checked)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Добавление разделителей и виджетов на панель

На панель инструментов можно добавлять не только действия, но и разделители (addSeparator()) и любые виджеты (addWidget()): 

toolbar.addAction(button_action)
toolbar.addSeparator()

toolbar.addWidget(QLabel("Текст:"))
toolbar.addWidget(QCheckBox())

Создание меню (Menus)

Главное меню создаётся через menuBar() у QMainWindow. Амперсанд (&) в названии меню задаёт клавишу быстрого доступа (Alt+буква).

menu_demo.py 
import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QLabel, QToolBar, QStatusBar
)
from PySide6.QtCore import Qt
from PySide6.QtGui import QAction

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("My App")
        
        label = QLabel("Привет!")
        label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(label)
        
        # Панель инструментов
        toolbar = QToolBar("Главная панель")
        self.addToolBar(toolbar)
        
        # Действие
        button_action = QAction("&Моя кнопка", self)
        button_action.setStatusTip("Это подсказка")
        button_action.triggered.connect(self.on_button_click)
        button_action.setCheckable(True)
        
        toolbar.addAction(button_action)
        self.setStatusBar(QStatusBar(self))
        
        # Меню
        menu = self.menuBar()
        file_menu = menu.addMenu("&Файл")
        file_menu.addAction(button_action)   # Переиспользуем действие!
        
        # Добавляем разделитель и второе действие
        file_menu.addSeparator()
        
        another_action = QAction("&Выход", self)
        another_action.triggered.connect(self.close)
        file_menu.addAction(another_action)
    
    def on_button_click(self, checked):
        print("Нажато!", checked)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Ключевая идея: Одно и то же действие button_action появляется и на панели инструментов, и в меню. Если действие переключаемое — оно остаётся таким везде.

Подменю (Submenus)

Подменю создаются вызовом addMenu() у существующего меню: 

file_menu = menu.addMenu("&Файл")
submenu = file_menu.addMenu("&Недавние")
submenu.addAction("project1.py")
submenu.addAction("project2.py")

Горячие клавиши (Shortcuts)

Горячая клавиша для действия задаётся методом setShortcut(). Qt поддерживает как текстовое описание ("Ctrl+S"), так и системно-независимые константы из QKeySequence: 

from PySide6.QtGui import QKeySequence

# Вариант 1: строка
button_action.setShortcut("Ctrl+P")

# Вариант 2: константа Qt
from PySide6.QtCore import Qt
button_action.setShortcut(Qt.CTRL | Qt.Key_P)

# Вариант 3: стандартная последовательность
button_action.setShortcut(QKeySequence.Print)

При использовании стандартных последовательностей (QKeySequence.Copy, QKeySequence.Paste и т.д.) Qt автоматически подставляет правильные сочетания клавиш для каждой платформы (Ctrl на Windows/Linux, Cmd на macOS).

Полный пример: панель инструментов + меню + статусная строка

full_example.py 
import sys
import os
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QLabel, QToolBar, QStatusBar,
    QCheckBox, QVBoxLayout, QWidget
)
from PySide6.QtCore import Qt, QSize
from PySide6.QtGui import QAction, QIcon, QKeySequence

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Моё приложение")
        
        # Центральный виджет
        self.label = QLabel("Привет!")
        self.label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(self.label)
        
        # Панель инструментов
        toolbar = QToolBar("Главная панель")
        toolbar.setIconSize(QSize(16, 16))
        self.addToolBar(toolbar)
        
        # Действие 1 (переключаемое)
        self.action1 = QAction("&Показать сообщение", self)
        self.action1.setStatusTip("Выводит сообщение в консоль")
        self.action1.triggered.connect(self.on_action_click)
        self.action1.setCheckable(True)
        self.action1.setShortcut(QKeySequence("Ctrl+M"))
        
        # Действие 2 (переключаемое)
        self.action2 = QAction("&Очистить метку", self)
        self.action2.setStatusTip("Очищает текст метки")
        self.action2.triggered.connect(self.clear_label)
        self.action2.setShortcut(QKeySequence("Ctrl+L"))
        
        toolbar.addAction(self.action1)
        toolbar.addSeparator()
        toolbar.addAction(self.action2)
        toolbar.addWidget(QCheckBox("Опция"))
        
        # Строка состояния
        self.setStatusBar(QStatusBar(self))
        
        # Меню
        menubar = self.menuBar()
        file_menu = menubar.addMenu("&Файл")
        file_menu.addAction(self.action1)
        file_menu.addAction(self.action2)
        file_menu.addSeparator()
        
        exit_action = QAction("&Выход", self)
        exit_action.triggered.connect(self.close)
        exit_action.setShortcut(QKeySequence.Quit)
        file_menu.addAction(exit_action)
        
        edit_menu = menubar.addMenu("&Правка")
        edit_menu.addAction(self.action1)
        
        # Подменю
        recent_menu = file_menu.addMenu("&Недавние")
        recent_menu.addAction("Документ 1")
        recent_menu.addAction("Документ 2")
    
    def on_action_click(self, checked):
        print("Действие активировано! Состояние:", checked)
        if checked:
            self.label.setText("Кнопка включена")
        else:
            self.label.setText("Кнопка выключена")
    
    def clear_label(self):
        self.label.setText("Метка очищена")

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
Итог лекции: QAction — это универсальный строительный блок для пользовательского ввода в Qt. Один раз определив действие (текст, иконку, горячую клавишу, подсказку), вы можете использовать его на панелях инструментов, в меню, в контекстных меню и в любых других местах. Это уменьшает дублирование кода и упрощает поддержку приложения.
