События (Events):
низкоуровневое взаимодействие

🖱️ Темы: QMouseEvent, mouse tracking, contextMenuEvent, QContextMenuEvent ⏱ ~30 мин

Что такое события в Qt?

Каждое взаимодействие пользователя с приложением Qt — движение мыши, нажатие клавиши, касание экрана — порождает событие. Qt представляет эти события с помощью специальных объектов-обёрток (например, QMouseEvent, QKeyEvent), которые содержат всю информацию о произошедшем.

События передаются в обработчики событий — специальные методы виджетов. Переопределяя эти методы, вы можете изменить стандартное поведение виджетов или добавить собственную логику.

Важно различать:
  • Сигналы — высокоуровневые уведомления (например, clicked, textChanged)
  • События — низкоуровневые примитивы (движение мыши, нажатие кнопки, нажатие клавиши)
Сигналы часто являются следствием событий, но они более удобны для повседневного использования.

Основные обработчики событий мыши

Обработчик событияТип события
mouseMoveEvent(self, e)Мышь перемещена над виджетом
mousePressEvent(self, e)Кнопка мыши нажата
mouseReleaseEvent(self, e)Кнопка мыши отпущена
mouseDoubleClickEvent(self, e)Двойное нажатие кнопки мыши
contextMenuEvent(self, e)Запрос на показ контекстного меню (обычно правый клик)
mouse_events_demo.py 
import sys
from PySide6.QtCore import Qt
from PySide6.QtWidgets import QApplication, QLabel, QMainWindow

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.label = QLabel("Нажмите, отпустите или переместите мышь в этом окне")
        self.label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(self.label)
        
        # Включаем отслеживание движения мыши без нажатой кнопки
        self.setMouseTracking(True)
    
    def mouseMoveEvent(self, e):
        self.label.setText(f"mouseMoveEvent: ({e.pos().x()}, {e.pos().y()})")
    
    def mousePressEvent(self, e):
        if e.button() == Qt.LeftButton:
            self.label.setText("mousePressEvent: LEFT")
        elif e.button() == Qt.RightButton:
            self.label.setText("mousePressEvent: RIGHT")
        else:
            self.label.setText("mousePressEvent: OTHER")
    
    def mouseReleaseEvent(self, e):
        self.label.setText("mouseReleaseEvent")
    
    def mouseDoubleClickEvent(self, e):
        self.label.setText("mouseDoubleClickEvent")

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
💡 Заметка: По умолчанию mouseMoveEvent срабатывает только при зажатой кнопке мыши. Чтобы получать события перемещения в любой момент, вызовите self.setMouseTracking(True).

Объект QMouseEvent — информация о событии

Внутри обработчика у вас есть доступ к объекту события e, который предоставляет следующие методы:

МетодВозвращаемое значение
.button()Конкретная кнопка, вызвавшая событие
.buttons()Состояние всех кнопок мыши (битовые флаги)
.pos()Позиция относительно виджета (QPoint)
.posF()Позиция с плавающей точкой (QPointF)
.globalPos()Глобальная позиция относительно экрана
.globalX(), .globalY()Глобальные координаты X и Y

Различение левой, правой и средней кнопок

mouse_buttons.py 
from PySide6.QtCore import Qt

def mousePressEvent(self, e):
    if e.button() == Qt.LeftButton:
        self.label.setText("Левая кнопка нажата")
    elif e.button() == Qt.RightButton:
        self.label.setText("Правая кнопка нажата")
    elif e.button() == Qt.MiddleButton:
        self.label.setText("Средняя кнопка нажата")
    else:
        self.label.setText("Другая кнопка")

def mouseReleaseEvent(self, e):
    if e.button() == Qt.LeftButton:
        self.label.setText("Левая кнопка отпущена")
Важно: В mouseDoubleClickEvent перед ним всегда происходит mousePressEvent и mouseReleaseEvent. Если вы обрабатываете оба события, учитывайте, что двойной клик вызовет последовательность: нажатие → отпускание → двойное нажатие → отпускание.

Отслеживание позиции мыши

mouse_position.py 
def mouseMoveEvent(self, e):
    # Локальные координаты (относительно виджета)
    local_x = e.pos().x()
    local_y = e.pos().y()
    
    # Глобальные координаты (относительно экрана)
    global_x = e.globalX()
    global_y = e.globalY()
    
    self.label.setText(f"Локально: ({local_x}, {local_y}) | Глобально: ({global_x}, {global_y})")

Контекстные меню (Context Menu)

Контекстное меню обычно появляется при щелчке правой кнопкой мыши. Qt предоставляет два способа его создания: через переопределение contextMenuEvent или через сигнал customContextMenuRequested.

Способ 1: переопределение contextMenuEvent

context_menu_override.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QMenu, QLabel
from PySide6.QtGui import QAction
from PySide6.QtCore import Qt

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Контекстное меню")
        self.label = QLabel("Правый клик для контекстного меню")
        self.label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(self.label)
    
    def contextMenuEvent(self, e):
        context_menu = QMenu(self)
        
        action1 = QAction("Действие 1", self)
        action1.triggered.connect(lambda: self.label.setText("Выбрано Действие 1"))
        
        action2 = QAction("Действие 2", self)
        action2.triggered.connect(lambda: self.label.setText("Выбрано Действие 2"))
        
        context_menu.addAction(action1)
        context_menu.addAction(action2)
        context_menu.addSeparator()
        context_menu.addAction("Выход", self.close)
        
        # Показываем меню в глобальной позиции курсора
        context_menu.exec(e.globalPos())

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()

Способ 2: через сигнал (более гибкий)

context_menu_signal.py 
from PySide6.QtCore import Qt

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Контекстное меню (сигнал)")
        
        label = QLabel("Правый клик для контекстного меню")
        label.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(label)
        
        # Включаем режим пользовательского контекстного меню
        self.setContextMenuPolicy(Qt.CustomContextMenu)
        # Подключаем сигнал к слоту
        self.customContextMenuRequested.connect(self.show_context_menu)
    
    def show_context_menu(self, pos):
        # pos — локальные координаты (относительно виджета)
        context_menu = QMenu(self)
        context_menu.addAction("Действие 1", lambda: print("Действие 1"))
        context_menu.addAction("Действие 2", lambda: print("Действие 2"))
        context_menu.addSeparator()
        context_menu.addAction("Выход", self.close)
        
        # mapToGlobal преобразует локальные координаты в глобальные
        context_menu.exec(self.mapToGlobal(pos))
Какой способ выбрать?
  • Переопределение contextMenuEvent — просто и наглядно
  • Сигнал customContextMenuRequested — удобнее, если нужно связать одно меню с несколькими виджетами или использовать меню из другого места приложения

Полный пример: рисование с помощью мыши

Создадим простой холст, на котором можно рисовать, перетаскивая мышь с зажатой левой кнопкой.

simple_paint.py 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QWidget
from PySide6.QtGui import QPainter, QPen, QColor
from PySide6.QtCore import Qt, QPoint

class PaintWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.setMouseTracking(True)
        self.last_point = None
        self.points = []  # Список нарисованных линий (список точек)
        self.current_line = []  # Текущая рисуемая линия
    
    def mousePressEvent(self, e):
        if e.button() == Qt.LeftButton:
            self.current_line = [e.pos()]
            self.last_point = e.pos()
    
    def mouseMoveEvent(self, e):
        if e.buttons() & Qt.LeftButton:  # Если зажата левая кнопка
            if self.last_point is not None:
                self.current_line.append(e.pos())
                self.last_point = e.pos()
                self.update()  # Перерисовываем виджет
    
    def mouseReleaseEvent(self, e):
        if e.button() == Qt.LeftButton and self.current_line:
            self.points.append(self.current_line.copy())
            self.current_line = []
            self.last_point = None
    
    def paintEvent(self, e):
        painter = QPainter(self)
        painter.setPen(QPen(QColor(94, 155, 255), 3, Qt.SolidLine))
        
        # Рисуем завершённые линии
        for line in self.points:
            for i in range(len(line) - 1):
                painter.drawLine(line[i], line[i + 1])
        
        # Рисуем текущую линию
        for i in range(len(self.current_line) - 1):
            painter.drawLine(self.current_line[i], self.current_line[i + 1])

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Рисование мышью")
        self.setCentralWidget(PaintWidget())
        self.resize(600, 400)

app = QApplication(sys.argv)
window = MainWindow()
window.show()
app.exec()
🎨 Результат: Приложение, в котором можно рисовать линии мышью с зажатой левой кнопкой. Это демонстрирует обработку последовательности событий: нажатие → движение → отпускание.

Полезные методы для работы с событиями

  • .accept() — принять событие (оно не будет передаваться дальше по иерархии виджетов)
  • .ignore() — игнорировать событие (родительский виджет тоже получит его)
  • setMouseTracking(True/False) — включить/выключить отслеживание движения мыши без нажатия
  • grabMouse() / releaseMouse() — захватить/освободить все события мыши
Итог лекции и всего курса:
  • События — фундамент, на котором строятся сигналы и взаимодействие с пользователем
  • Переопределяя обработчики событий, вы можете создавать любое кастомное поведение
  • Контекстные меню легко реализуются через contextMenuEvent или сигнал
  • Комбинация событий мыши, клавиатуры и таймеров позволяет создавать сложные интерактивные приложения
